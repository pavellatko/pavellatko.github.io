for fn in be vr nh an sd; do
    SECONDS=0
    grompp -f "$fn.mdp" -c etane.gro -p et.top -o "et_$fn.tpr" > /dev/null 2>&1
    mdrun -deffnm "et_$fn" -v -nt 1 > /dev/null 2>&1
    echo 2 | trjconv -f "et_$fn.trr" -s "et_$fn.tpr" -o "et_$fn.pdb" > /dev/null 2>&1
    duration=$SECONDS
    echo "$fn $duration seconds"
done;

    t = a[:,0]
    y = a[:, 1]
    plt.plot(t, y)

Как видим, в методах Velocity rescale и стохастической молекулярной динамики графики получились довольно похожи, энергия колеблется около одного значения, разброс не очень большой. У метода Андерсена разбрс совсем маленький, но значение энергии подозрительно меньше остальных методоы. У Нуза-Хувера лишком большой разброс, от 0 до более, чем 300, у Берендсена амплитуда постепенно уменьшается.

Таким образом наилучгие результаты дают Velocity rescale и метод Андерсена.
