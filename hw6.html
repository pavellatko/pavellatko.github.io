template = "C#CC1=CC=C(C=C1)C(C)C(=O)O"
found_newmols = []
for smi in smiles[:1500]:
    if "N=[N+]=[N-]" in smi:
        newsmi = smi.replace('N=[N+]=[N-]', template)
    else:
        continue
    try:
        newmol = Chem.MolFromSmiles(newsmi)
        donors = Lipinksy.NumHDonors(newmol)
        acceptors = Lipinksy.NumHAcceptors(newmol)
        weight = Lipinksy.rdMolDescriptors.CalcExactMolWt(newmol)
        log_p = Lipinksy.rdMolDescriptors.CalcCrippenDescriptors(newmol)[0]
        if donors <=5 and acceptors <= 10 and weight < 500 and log_p <= 5:
            found_newmols.append(newmol)
    except:
        pass
